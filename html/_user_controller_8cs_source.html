156 {

158 var key = Encoding.ASCII.GetBytes("veryverysecret.....");
159
160
161 var identity = new ClaimsIdentity(new Claim[]

163 new Claim(ClaimTypes.Role, user.Role),

166 });

169
170 var tokenDescriptor = new SecurityTokenDescriptor
171 {
172 Subject = identity,
173 Expires = DateTime.Now.AddDays(1),
174 SigningCredentials = credentials
175 };
176
177 var token = jwtTokenHandler.CreateToken(tokenDescriptor);
178 return jwtTokenHandler.WriteToken(token);
